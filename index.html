lib/features/cart/bloc/cart_bloc.dart Dart 
// ─── CartBloc ──────────────────────────────────────────────────
import 'package:flutter_bloc/flutter_bloc.dart';
import '../repository/cart_repository.dart';
import 'cart_event.dart'; import 'cart_state.dart';

class CartBloc extends Bloc<CartEvent, CartState> {
  final CartRepository _repository;

  CartBloc({required CartRepository repository})
      : _repository = repository,
        super(CartInitial()) {
    on<AddToCartEvent>(_onAddToCart);
    on<RemoveFromCartEvent>(_onRemoveFromCart);
    on<ApplyDiscountEvent>(_onApplyDiscount);
  }

  Future<void> _onAddToCart(
    AddToCartEvent event,
    Emitter<CartState> emit,
  ) async {
    emit(CartLoading());
    try {
      final updatedCart = await _repository.addItem(
        item: event.item,
        size: event.selectedSize,
        extras: event.extras,
      );
      emit(CartLoaded(cart: updatedCart));
    } catch (e) {
      emit(CartError(message: e.toString()));
    }
  }

  Future<void> _onApplyDiscount(
    ApplyDiscountEvent event,
    Emitter<CartState> emit,
  ) async {
    if (state is! CartLoaded) return;
    final current = (state as CartLoaded).cart;
    final discounted = await _repository
        .applyDiscount(cart: current, code: event.code);
    emit(CartLoaded(cart: discounted));
  }
}

// ─── Cart Events ───────────────────────────────────────────────
abstract class CartEvent {}

/// Fired when user taps "Add to Cart" with size + extras.
class AddToCartEvent extends CartEvent {
  final CartItem      item;
  final String        selectedSize;
  final List<Extra>   extras;

  AddToCartEvent({
    required this.item,
    required this.selectedSize,
    this.extras = const [],
  });
}

/// Removes a single item from the cart by ID.
class RemoveFromCartEvent extends CartEvent {
  final String itemId;
  RemoveFromCartEvent({required this.itemId});
}

/// Applies a promo/discount code to the active cart.
class ApplyDiscountEvent extends CartEvent {
  final String code;
  ApplyDiscountEvent({required this.code});
}

class ClearCartEvent extends CartEvent {}
lib/features/cart/bloc/cart_state.dart Dart 
// ─── Cart States ───────────────────────────────────────────────
abstract class CartState {}

class CartInitial extends CartState {}
class CartLoading extends CartState {}

class CartLoaded extends CartState {
  final Cart cart;
  const CartLoaded({required this.cart});

  /// Subtotal minus any applied discount.
  double get totalAfterDiscount =>
      cart.subtotal - cart.discountAmount;

  /// Sum of all item quantities in the cart.
  int get itemCount => cart.items.fold(
        0,
        (sum, item) => sum + item.quantity,
      );
}

class CartError extends CartState {
  final String message;
  CartError({required this.message});
}
